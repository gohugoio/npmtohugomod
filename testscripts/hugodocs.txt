# Packages here have their ESM build at the top level of dist/ (e.g.
# dist/module.esm.js) rather than in a dist/esm/ subdirectory, so we fall back
# to mounting the whole dist/ directory based on the package.json "module" field.
#
# Note also that an unscoped "alpinejs" and a scoped "@alpinejs/X" coexist as
# nested Hugo/Go modules ("@" is stripped from module paths).

npmtohugomod --module-base example.com/test

! stderr .

exists alpinejs/focus/package/dist/module.esm.js
exists alpinejs/persist/package/dist/module.esm.js
exists hotwired/turbo/package/dist/turbo.es2017-esm.js
exists alpinejs/package/dist/module.esm.js

cmp alpinejs/focus/hugo.toml golden/alpinejs/focus/hugo.toml
cmp alpinejs/persist/hugo.toml golden/alpinejs/persist/hugo.toml
cmp hotwired/turbo/hugo.toml golden/hotwired/turbo/hugo.toml
cmp alpinejs/hugo.toml golden/alpinejs/hugo.toml

# go.mod files are created by `hugo mod init`. Assert each module line.
grep '^module example.com/test/alpinejs/focus/v3$' alpinejs/focus/go.mod
grep '^module example.com/test/alpinejs/persist/v3$' alpinejs/persist/go.mod
grep '^module example.com/test/hotwired/turbo/v8$' hotwired/turbo/go.mod
grep '^module example.com/test/alpinejs/v3$' alpinejs/go.mod

# Idempotency: re-running with the same package.json should be a no-op
# (in particular it must not fail on `hugo mod init` complaining about an
# existing go.mod).
npmtohugomod --module-base example.com/test
! stderr .
grep '"version": "v3.15.3"' alpinejs/npmpackage.json

# Version bump: change alpinejs from 3.15.3 to 3.15.12 and re-run. The bumped
# package must update; the other (unchanged) dependencies must be left alone.
cp package-bumped.json package.json
npmtohugomod --module-base example.com/test
! stderr .

grep '"version": "v3.15.12"' alpinejs/npmpackage.json
exists alpinejs/package/dist/module.esm.js
exists alpinejs/go.mod

grep '"version": "v3.15.3"' alpinejs/focus/npmpackage.json
grep '"version": "v3.15.3"' alpinejs/persist/npmpackage.json
grep '"version": "v8.0.20"' hotwired/turbo/npmpackage.json
